def get_cs
  sum = 0
  [start, len, command, year, month, day, hour, minute, second].each do |n|
    sum = sum + sumit(n)
  end
  hex(((sum)&0xff))
end

def hex(v, string=true)
  unless string
    return v.to_i.to_s(16).upcase
  end
  '0x'+v.to_i.to_s(16).upcase
end

def le2be(v)
  a = ("%04x" % v.to_i.to_s(2).to_i(2)).scan(/./)
  (a[2]+a[3]+a[0]+a[1]).to_i(16)
end
